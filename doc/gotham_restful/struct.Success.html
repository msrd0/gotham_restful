pub struct Success<T> { /* private fields */ }
Expand description

This can be returned from a resource when there is no cause of an error.

Usage example:

#[derive(Deserialize, Serialize)]
struct MyResponse {
	message: &'static str
}

#[read_all]
fn read_all() -> Success<MyResponse> {
	let res = MyResponse {
		message: "I'm always happy"
	};
	res.into()
}
