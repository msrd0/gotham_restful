21/// Specify the allowed origins of the request. It is up to the browser to check the validity of the
22/// origin. This, when sent to the browser, will indicate whether or not the request's origin was
23/// allowed to make the request.
24#[derive(Clone, Debug)]
25pub enum Origin {
26	/// Do not send any `Access-Control-Allow-Origin` headers.
27	None,
28	/// Send `Access-Control-Allow-Origin: *`. Note that browser will not send credentials.
29	Star,
30	/// Set the `Access-Control-Allow-Origin` header to a single origin.
31	Single(String),
32	/// Copy the `Origin` header into the `Access-Control-Allow-Origin` header.
33	Copy
34}
35
36impl Default for Origin {
37	fn default() -> Self {
38		Self::None
39	}
40}
41
42impl Origin {
43	/// Get the header value for the `Access-Control-Allow-Origin` header.
44	fn header_value(&self, state: &State) -> Option<HeaderValue> {
45		match self {
46			Self::None => None,
47			Self::Star => Some("*".parse().unwrap()),
48			Self::Single(origin) => Some(origin.parse().unwrap()),
49			Self::Copy => {
50				let headers = HeaderMap::borrow_from(state);
51				headers.get(ORIGIN).cloned()
52			}
53		}
54	}
55
56	/// Returns true if the `Vary` header has to include `Origin`.
57	fn varies(&self) -> bool {
58		matches!(self, Self::Copy)
59	}
60}

62/// Specify the allowed headers of the request. It is up to the browser to check that only the allowed
63/// headers are sent with the request.
64#[derive(Clone, Debug)]
65pub enum Headers {
66	/// Do not send any `Access-Control-Allow-Headers` headers.
67	None,
68	/// Set the `Access-Control-Allow-Headers` header to the following header list. If empty, this
69	/// is treated as if it was [None].
70	List(Vec<HeaderName>),
71	/// Copy the `Access-Control-Request-Headers` header into the `Access-Control-Allow-Header`
72	/// header.
73	Copy
74}
75
76impl Default for Headers {
77	fn default() -> Self {
78		Self::None
79	}
80}
81
82impl Headers {
83	/// Get the header value for the `Access-Control-Allow-Headers` header.
84	fn header_value(&self, state: &State) -> Option<HeaderValue> {
85		match self {
86			Self::None => None,
87			Self::List(list) => Some(list.join(",").parse().unwrap()),
88			Self::Copy => {
89				let headers = HeaderMap::borrow_from(state);
90				headers.get(ACCESS_CONTROL_REQUEST_HEADERS).cloned()
91			}
92		}
93	}
94
95	/// Returns true if the `Vary` header has to include `Origin`.
96	fn varies(&self) -> bool {
97		matches!(self, Self::Copy)
98	}
99}

101/// This is the configuration that the CORS handler will follow. Its default configuration is basically
102/// not to touch any responses, resulting in the browser's default behaviour.
103///
104/// To change settings, you need to put this type into gotham's [State]:
105///
106/// ```rust,no_run
107/// # use gotham::{router::builder::*, pipeline::*, state::State};
108/// # use gotham_restful::{*, cors::Origin};
109/// # #[cfg_attr(feature = "cargo-clippy", allow(clippy::needless_doctest_main))]
110/// fn main() {
111/// 	let cors = CorsConfig {
112/// 		origin: Origin::Star,
113/// 		..Default::default()
114/// 	};
115/// 	let (chain, pipelines) = single_pipeline(new_pipeline().add(cors).build());
116/// 	gotham::start(
117/// 		"127.0.0.1:8080",
118/// 		build_router(chain, pipelines, |route| {
119/// 			// your routing logic
120/// 		})
121/// 	);
122/// }
123/// ```
124///
125/// This easy approach allows you to have one global cors configuration. If you prefer to have separate
126/// configurations for different scopes, you need to register the middleware inside your routing logic:
127///
128/// ```rust,no_run
129/// # use gotham::{router::builder::*, pipeline::*, state::State};
130/// # use gotham_restful::{*, cors::Origin};
131/// let pipelines = new_pipeline_set();
132///
133/// // The first cors configuration
134/// let cors_a = CorsConfig {
135/// 	origin: Origin::Star,
136/// 	..Default::default()
137/// };
138/// let (pipelines, chain_a) = pipelines.add(new_pipeline().add(cors_a).build());
139///
140/// // The second cors configuration
141/// let cors_b = CorsConfig {
142/// 	origin: Origin::Copy,
143/// 	..Default::default()
144/// };
145/// let (pipelines, chain_b) = pipelines.add(new_pipeline().add(cors_b).build());
146///
147/// let pipeline_set = finalize_pipeline_set(pipelines);
148/// gotham::start(
149/// 	"127.0.0.1:8080",
150/// 	build_router((), pipeline_set, |route| {
151/// 		// routing without any cors config
152/// 		route.with_pipeline_chain((chain_a, ()), |route| {
153/// 			// routing with cors config a
154/// 		});
155/// 		route.with_pipeline_chain((chain_b, ()), |route| {
156/// 			// routing with cors config b
157/// 		});
158/// 	})
159/// );
160/// ```
161#[derive(Clone, Debug, Default, NewMiddleware, StateData)]
162pub struct CorsConfig {
163	/// The allowed origins.
164	pub origin: Origin,
165	/// The allowed headers.
166	pub headers: Headers,
167	/// The amount of seconds that the preflight request can be cached.
168	pub max_age: u64,
169	/// Whether or not the request may be made with supplying credentials.
170	pub credentials: bool
171}

173impl Middleware for CorsConfig {
174	fn call<Chain>(self, mut state: State, chain: Chain) -> Pin<Box<HandlerFuture>>
175	where
176		Chain: FnOnce(State) -> Pin<Box<HandlerFuture>>
177	{
178		state.put(self);
179		chain(state)
180	}
181}

183/// Handle CORS for a non-preflight request. This means manipulating the `res` HTTP headers so that
184/// the response is aligned with the `state`'s [CorsConfig].
185///
186/// If you are using the [Resource](crate::Resource) type (which is the recommended way), you'll never
187/// have to call this method. However, if you are writing your own handler method, you might want to
188/// call this after your request to add the required CORS headers.
189///
190/// For further information on CORS, read <https://developer.mozilla.org/en-US/docs/Web/HTTP/CORS>.
191pub fn handle_cors(state: &State, res: &mut Response<Body>) {
192	let config = CorsConfig::try_borrow_from(state);
193	if let Some(cfg) = config {
194		let headers = res.headers_mut();
195
196		// non-preflight requests require the Access-Control-Allow-Origin header
197		if let Some(header) = cfg.origin.header_value(state) {
198			headers.insert(ACCESS_CONTROL_ALLOW_ORIGIN, header);
199		}
200
201		// if the origin is copied over, we should tell the browser by specifying the Vary header
202		if cfg.origin.varies() {
203			let vary = headers
204				.get(VARY)
205				.map(|vary| format!("{},origin", vary.to_str().unwrap()));
206			headers.insert(VARY, vary.as_deref().unwrap_or("origin").parse().unwrap());
207		}
208
209		// if we allow credentials, tell the browser
210		if cfg.credentials {
211			headers.insert(
212				ACCESS_CONTROL_ALLOW_CREDENTIALS,
213				HeaderValue::from_static("true")
214			);
215		}
216	}
217}

219/// Add CORS routing for your path. This is required for handling preflight requests.
220///
221/// Example:
222///
223/// ```rust,no_run
224/// # use gotham::{hyper::{Body, Method, Response}, router::builder::*};
225/// # use gotham_restful::*;
226/// build_simple_router(|router| {
227/// 	// The handler that needs preflight handling
228/// 	router.post("/foo").to(|state| {
229/// 		let mut res: Response<Body> = unimplemented!();
230/// 		handle_cors(&state, &mut res);
231/// 		(state, res)
232/// 	});
233/// 	// Add preflight handling
234/// 	router.cors("/foo", Method::POST);
235/// });
236/// ```
237pub trait CorsRoute<C, P>
238where
239	C: PipelineHandleChain<P> + Copy + Send + Sync + 'static,
240	P: RefUnwindSafe + Send + Sync + 'static
241{
242	/// Handle a preflight request on `path` for `method`. To configure the behaviour, use
243	/// [CorsConfig].
244	fn cors(&mut self, path: &str, method: Method);
245}
246
247pub(crate) fn cors_preflight_handler(state: State) -> (State, Response<Body>) {
248	let config = CorsConfig::try_borrow_from(&state);
249
250	// prepare the response
251	let mut res = create_empty_response(&state, StatusCode::NO_CONTENT);
252	let headers = res.headers_mut();
253	let mut vary: Vec<HeaderName> = Vec::new();
254
255	// copy the request method over to the response
256	let method = HeaderMap::borrow_from(&state)
257		.get(ACCESS_CONTROL_REQUEST_METHOD)
258		.unwrap()
259		.clone();
260	headers.insert(ACCESS_CONTROL_ALLOW_METHODS, method);
261	vary.push(ACCESS_CONTROL_REQUEST_METHOD);
262
263	if let Some(cfg) = config {
264		// if we allow any headers, copy them over
265		if let Some(header) = cfg.headers.header_value(&state) {
266			headers.insert(ACCESS_CONTROL_ALLOW_HEADERS, header);
267		}
268
269		// if the headers are copied over, we should tell the browser by specifying the Vary header
270		if cfg.headers.varies() {
271			vary.push(ACCESS_CONTROL_REQUEST_HEADERS);
272		}
273
274		// set the max age for the preflight cache
275		if let Some(age) = config.map(|cfg| cfg.max_age) {
276			headers.insert(ACCESS_CONTROL_MAX_AGE, age.into());
277		}
278	}
279
280	// make sure the browser knows that this request was based on the method
281	headers.insert(VARY, vary.join(",").parse().unwrap());
282
283	handle_cors(&state, &mut res);
284	(state, res)
285}

287impl<D, C, P> CorsRoute<C, P> for D
288where
289	D: DrawRoutes<C, P>,
290	C: PipelineHandleChain<P> + Copy + Send + Sync + 'static,
291	P: RefUnwindSafe + Send + Sync + 'static
292{
293	fn cors(&mut self, path: &str, method: Method) {
294		let matcher = AccessControlRequestMethodMatcher::new(method);
295		self.options(path)
296			.extend_route_matcher(matcher)
297			.to(cors_preflight_handler);
298	}
